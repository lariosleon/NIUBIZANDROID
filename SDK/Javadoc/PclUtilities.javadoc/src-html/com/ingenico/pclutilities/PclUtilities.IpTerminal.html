178         * @param logMessage
179         *            The message to add to the log.

181        private static void logToFile(int level, String logMessageTag, String logMessage) {
182                if (mAppContext == null)
183                        return;
184                synchronized (mFileLock) {
185                        FileWriter fw = null;
186                        File logFile = null;
187                        try {
188                                // Gets the log file from the root of the primary storage. If it
189                                // does
190                                // not exist, the file is created.
191
192                                String filePath = mAppContext.getExternalFilesDir(null) + "/" + FILE_NAME;
193
194                                if(mLogPathFolder != null){
195                                        filePath = mLogPathFolder + "/" + FILE_NAME;
196                                }
197
198                                logFile = new File(filePath);
199                                if (!logFile.exists())
200                                        logFile.createNewFile();
201
202                                // Write the message to the log with a timestamp
203                                fw = new FileWriter(logFile, true);
204                                fw.write(String.format("%1s %2s %3s %4s\r\n", getDateTimeStamp(), getLevel(level), logMessageTag,
205                                                logMessage));
206
207                        } catch (IOException e) {
208                                Log.e(logMessageTag, "Unable to log exception to file: " + logMessageTag + " " + logMessage);
209                                e.printStackTrace();
210                        } finally {
211                                if(fw != null){
212                                        try{
213                                                fw.close();
214                                        } catch (IOException e) {
215                                                e.printStackTrace();
216                                        }
217                                }
218                        }
219                        changeLogFile(logFile);
220                }
221
222        }

1542        private UsbCompanion detectIngenicoDevice(UsbDevice device, UsbSerialPort port) {
1543                PclLog.d(TAG, "detectIngenicoDevice");
1544                UsbCompanion companion = null;
1545                if(device != null){
1546                        //call method to set up device communication
1547                        if (!mUsbManager.hasPermission(device)) {
1548                                //Log.w(TAG, "detectIngenicoDevice hasPermission returns false" + device);
1549                                notifyUsbErrorEvent(UsbError.Type.USB_HAS_NOT_PERMISSION,device, null);
1550                        } else {
1551                                UsbDeviceConnection connection = mUsbManager.openDevice(device);
1552                                byte buffer[] = new byte[64];
1553                                int totalRead = 0;
1554                                int read = 0, written = 0;
1555                                
1556                                String szAnswer = new String();
1557                                if (connection != null) {
1558                                        if (port != null) {
1559                                                try {
1560                                                        port.open(connection);
1561                                                        PclLog.i(TAG, "detectIngenicoDevice: Purge buffers");
1562                                                        // Read to purge buffers
1563                                                        read = port.read(buffer,  10);
1564                                                        totalRead += read;
1565                                                        if (read > 0)
1566                                                        {
1567                                                                do {
1568                                                                        // Wait more than 1s in case we receive LCP echo frames after lcp_close
1569                                                                        read = port.read(buffer,  1100);
1570                                                                        totalRead += read;
1571                                                                        //PclLog.d(TAG, "detectIngenicoDevice purge=" + read);
1572                                                                } while (read > 0 && totalRead < 256); // If total_read > 256 it probably means that the traces are activated on the terminal
1573                                                        }
1574                                                        port.purgeHwBuffers(true, true);
1575                                                        written = port.write(ATI.getBytes(), 1000);
1576                                                        //PclLog.d(TAG, "detectIngenicoDevice written=" + written);
1577                                                        
1578                                                        totalRead = 0;
1579                                                        do {
1580                                                                long time = System.currentTimeMillis();
1581                                                                read = port.read(buffer,  1000);
1582                                                                if (read == 0){
1583                                                                        // mesure duration for read return to detect driver error returning immediately
1584                                                                        if((System.currentTimeMillis()-time) < 10){
1585                                                                                notifyUsbErrorEvent(UsbError.Type.USB_CAN_NOT_READ, device, null);
1586                                                                        } else {
1587                                                                                notifyUsbErrorEvent(UsbError.Type.USB_NO_RESPONSE, device, null);
1588                                                                        }
1589                                                                        break;
1590                                                                }
1591                                                                szAnswer = szAnswer.concat(new String(buffer, 0, read));
1592                                                                //PclLog.d(TAG, "detectIngenicoDevice read=" + read);
1593                                                                //PclLog.d(TAG, szAnswer);
1594                                                                totalRead += read;
1595                                                        } while (!szAnswer.endsWith("\r\n\r\nOK\r\n") && totalRead < 256); // If total_read > 256 it probably means that the traces are activated on the terminal
1596                                                        if(totalRead > 256)
1597                                                        {
1598                                                                notifyUsbErrorEvent(UsbError.Type.USB_RESPONSE_TOO_LONG, device, "total read: "+totalRead);
1599                                                        }
1600                                                } catch (IOException e) {
1601                                                        e.printStackTrace();
1602                                                } catch ( StringIndexOutOfBoundsException e) {
1603                                                        e.printStackTrace();
1604                                                } finally {
1605                                                        try {
1606                                                                port.close();
1607                                                        } catch (IOException e) {                                               
1608                                                                e.printStackTrace();
1609                                                        }
1610                                                }
1611                                        }
1612                                        connection.close();
1613                                }
1614                                
1615                                if (totalRead > 10)
1616                                {
1617                                        if (szAnswer.startsWith("\r\n") && szAnswer.endsWith("\r\n\r\nOK\r\n"))
1618                                        {
1619                                                String szName = szAnswer.substring(2, totalRead-8);
1620                                                companion = new UsbCompanion(device,  szName, port);
1621                                        }
1622                                }
1623                        }
1624                }
1625
