001package com.ingenico.pclutilities;
002
003import java.io.File;
004import java.io.Serializable;
005
006/**
007 * 
008 * Class to manage certificate used by the application to establish ssl connection
009 *
010 */
011public class SslObject implements Serializable {
012
013        public static final String ACTION_SSL_NOTIFY = "com.ingenico.pclservice.intent.action.SSL_NOTIFY";
014        public static final String EXTRA_SSL_STATE= "SSL_STATE";
015        private static final long serialVersionUID = 1L;
016        private String mPackageName = null;
017        private String mName = null;
018        private String mPassword = null;
019        private String mOldPath = null;
020
021        /**
022         * Create ssl file informations Object to enable PCL service to read the content
023         * @param packageName: private context containing the ssl file
024         * @param file: ssl file name
025         * @param password: password of ssl file (null if none)
026         */
027        public SslObject(String packageName, String file, String password){
028                this.mPackageName = packageName;
029                this.mName = file;
030                this.mPassword = password;
031        }
032
033        /**
034         * @deprecated Create ssl file informations Object to enable PCL service to read the content
035         * @param file: existing file containing ssl object
036         * @param password: password of ssl file (null if none)
037         */
038        public SslObject(File file, String password){
039                this.mOldPath = file.getPath();
040                this.mPassword = password;
041        }
042
043        /**
044         * @deprecated Get the file path of ssl file
045         * @return The file path
046         */
047        public String getPath(){ return mOldPath;}
048
049        /**
050         * @deprecated Save the file path of ssl file
051         * @param file: existing file containing ssl object
052         */
053        public void setFile(File file) { mOldPath = file.getPath();}
054
055        /**
056         * Get the file name of the ssl file
057         * @return The file name
058         */
059        public String getName() {
060                return mName;
061        }
062
063        /**
064         * Save name of ssl file
065         * @param name: name of existing file containing ssl object
066         */
067        public void setName(String name) {
068                this.mName = name;
069        }
070
071        /**
072         * Get the password to read the ssl file
073         * @return The password
074         */
075        public String getPassword() {
076                return mPassword;
077        }
078
079        /**
080         * Save the password of the ssl file
081         * @param password: The password  (null if none)
082         */
083        public void setPassword(String password) {
084                this.mPassword = password;
085        }
086
087        /**
088         * Get the package name of private context where the ssl file exist
089         * @return The package name
090         */
091        public String getPackage() {
092                return mPackageName;
093        }
094
095        /**
096         * Save the package name of private context where the ssl file exist
097         * @param packageName: The package name
098         */
099        public void setPackage(String packageName) {
100                this.mPackageName = packageName;
101        }
102}
