197 // Fill the field with 0 left justified
198    private void FillWithZero(byte[] dest, String src)
199    {
200        int i;
201        byte[] srcByteArray = src.getBytes(Charset.defaultCharset());
202        for (i=0; i<dest.length; i++)
203                dest[i] = '0';
204        for (i=dest.length - src.length(); i<dest.length; i++) {
205                dest[i] = srcByteArray[i-dest.length+src.length()];
206        }
207    }
208    
209    private boolean setData(byte[] dest, int size, String src) {
210        if (src.length() > size)
211                return false;
212        FillWithZero(dest, src);
213        return true;
214    }
