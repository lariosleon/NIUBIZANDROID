001package com.ingenico.pclservice;
002
003import android.os.Binder;
004
005/**
006 * The PclBinder class enable to get PCL object after binding service
007 */
008public class PclBinder extends Binder {
009    private PclService service;
010    PclBinder(PclService service)
011    {
012        this.service = service;
013    }
014
015    /**
016     * @return running PCL service object
017     */
018    public PclService getService() {
019        return this.service;
020    }
021}
