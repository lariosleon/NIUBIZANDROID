005/**
006         * Gets server status.
007         *

009         * @return      always true

011boolean serverStatus(byte[] result);
012/**
013     * Open printer on remote device. 
014     *
015     * @param result  Result of the operation (0=OK, 1=KO)
016     * @return        true if the method succeeded (see result for command result).
017     *                false if the method failed.
018     */
019boolean openPrinter(byte[] result);
020/**
021     * Close printer on remote device. 
022     *
023     * @param result  Result of the operation (0=OK, 1=KO)
024     * @return        true if the method succeeded (see result for command result).
025     *                false if the method failed.
026     */
027boolean closePrinter(byte[] result);
028/**
029     * Print text on remote device. 
030     *
031     * @param strText  Null-terminated string with text to be printed. Max length is 511 characters. 
032     * The string must end with line feed character.
033     * @param result   Result of the operation (0=OK, 1=KO)
034     * @return         true if the method succeeded (see result for command result).
035     *                 false if the method failed.
036     */
037boolean printText(java.lang.String strText, byte[] result);
038/**
039     * Print bitmap on remote device.
040     * Bitmap characteristics are:
041     * -   max width: 384
042     * -   max height: 1024
043     * -   no compression
044     * -   1, 2 or 4 color bits 
045     *
046     * @param bmpBuf  Bitmap buffer
047     * @param bmpSize Bitmap size
048     * @param result  Result of the operation (0=OK, 1=KO)
049     * @return        true if the method succeeded (see result for command result).
050     *                false if the method failed.
051     */
052boolean printBitmap(byte[] bmpBuf, int bmpSize, byte[] result);
053/**
054     * Store logo (bitmap image) in remote device for further printing.
055     * Bitmap characteristics are:
056     * -   max width: 384
057     * -   max height: 1024
058     * -   no compression
059     * -   1, 2 or 4 color bits
060     *
061     * @param name    Null-terminated string containing the name of the logo
062     * (8 characters max, upper case)
063     * @param type    Unused. Must be set to 1.
064     * @param bmpBuf  Bitmap buffer
065     * @param bmpSize Bitmap size
066     * @param result  Result of the operation (0=OK, 1=KO)
067     * @return        true if the method succeeded (see result for command result).
068     *                false if the method failed.
069     */
070boolean storeLogo(java.lang.String name, int type, byte[] bmpBuf, int bmpSize, byte[] result);
071/**
072     * Print logo (bitmap image) stored in iWL (either with previous call to 
073     * storeLogo or with other available methods used to load some files in 
074     * terminal: LLT, Ingestate).
075     *
076     * @param name    Null-terminated string containing the name of the logo 
077     * (8 characters max, upper case)
078     * @param result  Result of the operation (0=OK, 1=KO)
079     * @return        true if the method succeeded (see result for command result).
080     *                false if the method failed.
081     */
082boolean printLogo(java.lang.String name, byte[] result);
083/**
084     * Get remote device printer status.
085     *
086     * @param result  Result of the operation (bitfield)
087     *                0x00=OK
088     *                0x01=KO
089     *                0x10=PRINTER ERROR
090     *                0x20=PAPER OUT
091     *                0x40=CONNECTION FAILED (when using Bluetooth printer)
092     *                0x80=PRINTER BATTERY LOW (when using Bluetooth printer)
093     * @return        true if the method succeeded (see result for command result).
094     *                false if the method failed.
095     */
096boolean getPrinterStatus(byte[] result);
097/**
098     * Get remote device time.

101     * @return        true if the method succeeded (see time for command result).
102     *                false if the method failed.
103     */
104boolean getTerminalTime(byte[] time);
105/**

107     *
108     * @param result  Result of the operation (0=OK, 1=KO)
109     * @return        true if the method succeeded (see result for command result).
110     *                false if the method failed.
111     */
112boolean setTerminalTime(byte[] result);
113/**
114     * Read Telium information (serial number, product number)

117     * @param productNumber  Product number of the remote device.
118     * @return        true if the method succeeded (see result for command result).
119     *                false if the method failed.
120     */
121boolean getTerminalInfo(byte[] serialNumber, byte[] productNumber);
122/**
123     * Fetch information about the components installed on the Telium terminal.
124     *
125     * @param strPath  File name where to store components list.
126     * The file will be stored in PclService data directory.
127     * You can access it with:
128     * createPackageContext("com.ingenico.pclservice",0).openFileInput(strPath);
129     * @return        true if the method succeeded (see result for command result).
130     *                false if the method failed.
131     */
132boolean getTerminalComponents(java.lang.String strPath);
133/**
134     * This launches a transaction request, passing the input parameters to the payment module.
135     * The output parameters are filled with the response from the payment module.
136     *
137     * @param transIn  Input parameters for transaction.
138     * @param transOut Output parameters for transaction.
139     * @return        true if the method succeeded
140     *                false if the method failed.
141     * @see TransactionIn
142     * @see TransactionOut
143     */
144boolean doTransaction(com.ingenico.pclservice.TransactionIn transIn, com.ingenico.pclservice.TransactionOut transOut);
145/**
146     * This launches a transaction request, passing the input parameters to the payment module
147     * and custom extended input data. The output parameters are filled with the response from 
148     * the payment module and custom extended output data are received too.
149     *
150     * @param transIn  Input parameters for transaction.
151     * @param transOut Output parameters for transaction.
152     * @param appNumber This value explicitly specified the Telium application number to contact
153     * (destination of the PDA_START_TRANSACTION ServiceCall, see Telium documentation for more detail).
154     * Set it to zero if no explicit selection is required.
155     * @param inBuffer      Pointer to the input extended data to pass as parameter to the PDA_START_TRANSACTION ServiceCall.
156     * @param inBufferSize  Size of the input extended data to pass as parameter to the PDA_START_TRANSACTION ServiceCall.
157     * @param outBuffer     Pointer to the output extended data to receive as parameter from the PDA_START_TRANSACTION ServiceCall.
158     * @param outBufferSize Maximum size of the output extended data to receive as parameter from the PDA_START_TRANSACTION ServiceCall, 
159     * and this variable is updated with the actual buffer size.
160     * @return        true if the method succeeded
161     *                false if the method failed.
162     * @throws IllegalArgumentException if outBuffer or outBufferSize is null
163     * @see TransactionIn
164     * @see TransactionOut
165     */
166boolean doTransactionEx(com.ingenico.pclservice.TransactionIn transIn, com.ingenico.pclservice.TransactionOut transOut, int appNumber, byte[] inBuffer, long inBufferSize, byte[] outBuffer, long[] outBufferSize) throws IllegalArgumentException;
167/**
