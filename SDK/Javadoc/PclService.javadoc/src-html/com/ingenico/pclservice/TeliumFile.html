001package com.ingenico.pclservice;
002
003import java.io.Serializable;
004
005/**
006 * Created by jpeltier on 19/01/2018.
007 */
008
009/**
010 * The TeliumFile class is a representation of a file get from a Telium device.
011 */
012public class TeliumFile implements Serializable {
013
014    protected String name;
015    protected String path;
016    protected Integer size;
017    protected Boolean directory;
018
019    TeliumFile() {
020        this.name = null;
021        this.path = null;
022        this.size = 0;
023        this.directory = false;
024    }
025
026    public TeliumFile(String name, String path, Integer size, Boolean directory) {
027        this.name = name;
028        this.path = path;
029        this.size = size;
030        this.directory = directory;
031    }
032
033    /**
034     * Get the name of the file.
035     *
036     * @return String Returns the name of the file.
037     */
038    public String getName() {
039        return this.name;
040    }
041
042    /**
043     * Get the path of the file on Telium.
044     *
045     * @return String Returns the path of the file on Telium.
046     */
047    public String getPath() {
048        return this.path;
049    }
050
051    /**
052     * Get the size of the file.
053     *
054     * @return Integer Returns the size of the file in bytes.
055     */
056    public Integer getSize() {
057        return this.size;
058    }
059
060    /**
061     * Check if the file is a directory.
062     *
063     * @return Boolean Returns true if the file is actually a directory.
064     *                 Returns false if the file is a file.
065     */
066    public Boolean isDirectory() {
067        return directory;
068    }
069
070    @Override
071    public String toString() {
072        String display = "TeliumFile:";
073
074        if (this.name != null) {
075            display += " "+this.name;
076        }
077
078        if (this.size != null) {
079            display += " ("+String.valueOf(this.size)+" bytes)";
080        }
081
082        if (this.path != null) {
083            display += " at "+this.path;
084        }
085
086        return display;
087    }
088}
